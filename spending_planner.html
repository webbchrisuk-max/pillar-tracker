function handleRollForward(e) { const btn = e.target.closest("[data-roll-section]"); if (!btn) return; const index = Number(btn.dataset.rollIndex); if (!Number.isFinite(index)) return; const bill = plannerState.scheduledBills[index]; if (!bill || !bill.paid) return; const surplus = Math.max(0, Number(bill.amount || 0) - Number(bill.actualPaid || 0)); if (surplus > 0) { plannerState.holdingBalance += surplus; } if (bill.due) { const d = new Date(bill.due); d.setMonth(d.getMonth() + 1); bill.due = d.toISOString().split("T")[0]; } bill.paid = false; bill.actualPaid = 0; bill.deducted = false; savePlannerData(); runPlanner(); }
